var
 nome, letra, maiorL: caractere     //maiorL = define a maior letra
 i, maior: inteiro
inicio
 maior <- 0        //maior começa com 0
 
 escreval("Escreva algo:") //entrada
   leia(nome)
    
 para i de 1 ate compr(nome) faca   //para ler a string
  letra <- copia(nome,i,1)
  i <- asc(letra)
  se(i > maior) entao    //se o caractere da string for maior, então maior o número ASCII e maiorL recebe o caractere
    maior <- i
    maiorL <- carac(i)
  fimse
 fimpara                 //fim para
 escreval("-------------------------------------------------")
 escreval("O maior caractere segundo a tabela ASCII é: ",maiorL)       //apresenta o maior caractere
