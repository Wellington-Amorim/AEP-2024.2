// EX20) Palíndromo. Um palíndromo é uma sequência de caracteres cuja leitura é
//idêntica se feita da direita para esquerda ou vice-versa. Faça um
//programa que leia uma string e informe ao usuário se o valor informado é
//um palíndromo ou não.
// Autor : Wellington Amorim de Sousa
// RA : 24214926-2
// Data : 08/09/2024
var
  nome,x,y: caractere
  i: inteiro                  //i = para fazer o "para"
inicio
  escreval("Escreva qualquer coisa:")     //entrada
    leia(nome)
    
  escreva("(")   //pra ficar bonitinho nna tela

  para i de 1 ate compr(nome) faca
    escreva(copia(nome, i, 1))
    x <- x + copia(nome, i, 1)
  fimpara

  escreva("/")  //pra ficar bonitinho nna tela

  para i de compr(nome) ate 1 passo -1 faca
    escreva(copia(nome, i, 1))
    y <- y + copia(nome, i, 1)
  fimpara
  
  escreval(")") //pra ficar bonitinho nna tela

  escreval("-------------------------------------") //pra ficar bonitinho nna tela
  se(x = y) entao
    escreval("É um palíndromo!")
  senao
    escreval("Não é um palíndromo!")
  fimse
  escreval("-------------------------------------") //pra ficar bonitinho nna tela
