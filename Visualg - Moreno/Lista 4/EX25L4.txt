var
  palavra, c, senha: caracter
  i, x: inteiro
inicio
  escreval("Escreva uma frase para obter uma senha: ")  //entrada
    leia(palavra)
    
 // senha <- ""
  para i de 1 ate compr(palavra) faca
    c <- copia(palavra,i,1)           //c = seleciona o caractere;
    x <- asc(c)                       //x = pega o valor na tabela ASCII;
    c <- carac(x+i+5)                 //soma da tabela ASCII, quantidade de caracteres e +5
    senha <- senha + c                //senha recebe o caractere modificado
  fimpara
  
  escreval("-------------------------------------")
  escreval(senha)          //saída
