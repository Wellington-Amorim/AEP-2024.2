// EX10)Solicite uma string (mínimo 5 e máximo 20 caracteres) e apresente-a ao
//usuário, intercalando cada caractere em maiúsculo e minúsculo.
// Autor : Wellington Amorim de Sousa
// RA : 24214926-2
// Data : 07/09/2024
var
  nome, c: caractere          //c = é o caractere
  i, x: inteiro                  //i = para fazer o "para"; x = faz o repita ate
inicio
  repita                        //loop
    escreval("Escreva qualquer coisa entre 5 e 20 caracteres:")     //entrada
      leia(nome)
      x <- compr(nome)
    escreval("---------------------------------")
  ate(x >= 5) e (x <= 20)       //fim loop

  para i de 1 ate compr(nome) faca
    c <- copia(nome,i,1)
    se(i%2 = 0) entao            //segue a mesma lógica do par e impar
      escreva(minusc(c))         //apresenta o caractere em minúsculo
    senao
      escreva(maiusc(c))         //apresenta o caractere em maiúculo
    fimse
  fimpara
fimalgoritmo
