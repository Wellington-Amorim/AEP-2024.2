// EX21) Palíndromo. Em textos mais complexos os espaços e pontuação são
//ignorados. A frase SUBI NO ONIBUS é o exemplo de uma frase
//palíndromo onde os espaços foram ignorados. Faça um programa que
//leia uma sequência de caracteres e informe ao usuário se a string
//informada é um palíndromo ou não.
// Autor : Wellington Amorim de Sousa
// RA : 24214926-2
// Data : 08/09/2024
var
  nome,x,y: caractere
  i: inteiro                  //i = para fazer o "para"
inicio
  escreval("Escreva qualquer coisa:")     //entrada
    leia(nome)
    
  escreva("(")   //pra ficar bonitinho na tela

  para i de 1 ate compr(nome) faca
    se(copia(nome, i, 1) <> " ") entao
      escreva(copia(nome, i, 1))
      x <- x + copia(nome, i, 1)     //x recebe a string da esquerda pra direita
    fimse
  fimpara

  escreva("/")  //pra ficar bonitinho na tela

  para i de compr(nome) ate 1 passo -1 faca
    se(copia(nome, i, 1) <> " ") entao
      escreva(copia(nome, i, 1))
      y <- y + copia(nome, i, 1)     //y recebe a string da direita pra esquerda
    fimse
  fimpara
  
  escreval(")") //pra ficar bonitinho na tela

  escreval("-------------------------------------") //pra ficar bonitinho na tela
  se(x = y) entao
    escreval("É um palíndromo!")
  senao
    escreval("Não é um palíndromo!")
  fimse
  escreval("-------------------------------------") //pra ficar bonitinho na tela
