  nome,inverso: caractere

inicio
  escreval("Escreva qualquer coisa:")     //entrada
    leia(nome)

  para i de compr(nome) ate 1 passo -1 faca   //para começar da quantidade de caracteres até 1, um paso para trás

  fimpara
