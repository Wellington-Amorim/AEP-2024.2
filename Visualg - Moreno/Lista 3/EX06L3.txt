// EX06) Solicite o nome, idade e sexo de uma pessoa. Valide o sexo entre("F"/"M"), idade entre 10 a 100 anos.
// Autor : Wellington Amorim de Sousa

// RA: 24214926-2
var
nome, sexo: caractere
idade: inteiro
inicio
escreval("Escreva o seu nome: ")   //entrada do nome
leia(nome)
escreval("Idade: ")         //entrada idade
leia(idade)
repita                  //repita idade
   se (idade < 10) ou (idade > 100) entao             //se idade for diferente do esperado
      escreval("A sua idade precisa ser entre 10 e 100 anos.")
      leia(idade)
   fimse                                          //fim do se idade difereente
ate(idade >= 10) e (idade <= 100)                 //fim do repita idade
escreval("Escolha o seu sexo(F/M): ")        //entrada sexo
leia(sexo)
repita                                            //repita sexo
   se (sexo <> "F") ou (sexo <> "M") entao     //se sexo for diferente de F ou M
      escreval("Digite F ou M.")
      leia(sexo)
   fimse                                     //fim se sexo diferente
ate(sexo = "F") ou (sexo = "M")          //fim repita sexo
escreval("Cadastro concluído.")
